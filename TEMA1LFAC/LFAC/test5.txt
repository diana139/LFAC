class Calculator
begin
    public
        function int computeHalf(int a)
        begin
            int result;
            result <- a / 2;      
            return result;
        end
        function void printDouble(int a)
        begin
            Print(a * 2);
        end
end
function bool par(int x)
begin 
    int a;
    verify (x%2==0) 
    begin
    return FALSE;
    end
    doesn't_verify begin return 1;
    end
end
